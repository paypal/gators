def compute_equal_size_bins(X: pl.DataFrame, num_bins: int) -> dict[str, list[float]]:
    """
    Discretizes numerical variables using an equal size bins.

    Parameters
    ----------
    X : pl.DataFrame
        Input DataFrame containing the data to discretize.
    num_bins : int
        Number of bins to divide each numeric column into.

    Returns
    -------
    dict[str, list[float]]
        Dictionary where keys are column names and values are lists of bin edges.

    Examples
    --------
    >>> import polars as pl
    >>> from gators.discretizers import compute_equal_size_bins
    >>> X = pl.DataFrame({
    ...     'A': [0.1, 0.2, 0.3, 0.4],
    ...     'B': [10, 20, 30, 40]
    ... })
    >>> bins = compute_equal_size_bins(X, num_bins=3)
    >>> print(bins)
    {'A': [0.2, 0.3], 'B': [20.0, 30.0]}
    """

    percentiles = np.linspace(0, 1, num_bins + 1)[1:-1]
    bins = X.select(
        [
            pl.col(col_name).quantile(p).alias(f"{col_name}_{p}")
            for p in percentiles
            for col_name in X.columns
        ]
    ).to_dict(as_series=False)
    selected_bins = {}
    for col in X.columns:
        bins_ = sorted(set([bins[f"{col}_{p}"][0] for p in percentiles]))  # unique bins
        bins_ = [b for b in bins_ if not isnan(b)]  # drop nan bins
        selected_bins[col] = bins_
    return selected_bins

    Equal-size discretizer.

    ----------
    subset : Optional[List[str]], default=None
        List of column names to discretize. If None, all numeric columns are used.

    rounding : PositiveInt, default=3

    >>> from gators.discretizers import EqualSizeDiscretizer

    def fit(
        self, X: pl.DataFrame, y: Optional[pl.Series] = None) -> "EqualSizeDiscretizer":
        """Fit the discretizer by computing equal-size (quantile-based) bin boundaries.

        Parameters
        ----------
        X : pl.DataFrame
            Input DataFrame with numeric columns.
        y : Optional[pl.Series], default=None
            Target series (not used, present for sklearn compatibility).

        Returns
        -------
        EqualSizeDiscretizer
            The fitted discretizer instance.
        """
        if not self.subset:
            self.subset = [
                col
                for col, dtype in zip(X.columns, X.dtypes)
                if dtype in [pl.Float64, pl.Int64, pl.Float32, pl.Int32]
            ]

        self._bins = compute_equal_size_bins(X[self.subset], self.num_bins)
        self._labels = generate_labels(self._bins)
        if self.as_numerics:
            self._labels = {
                col: np.arange(len(vals)).astype(str)
                for col, vals in self._labels.items()
            }
        if not self.inplace:
            self._column_mapping = {
                col: f"{col}__dicretize_size" for col in self.subset
            }
        return self
