data = pd.read_parquet('data/sf_crime.parquet')
data = data.reset_index(drop=True)

def f1_score_binary(y_true, y_pred):
    p = y_true[y_pred == 1].mean()
    r = y_pred[y_true == 1].mean()
    if (p == 0) | (r == 0):
        return 0
    return 2 * p * r / (p + r)
f1_scoring = make_scorer(f1_score_binary)
